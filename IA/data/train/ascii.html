// Fonction pour charger la clé d'API depuis un fichier de configuration
async function loadApiKey() {
  const response = await fetch('../inc/config.php');
  const config = await response.json();
  const api_key = config.api_key;
  return api_key;
}

// Fonction pour effectuer la requête à l'API ChatGPT
async function generateChatGPTMessage(prompt) {
  const api_key = await loadApiKey(); // Charger la clé d'API
  const endpoint = 'https://api.openai.com/v1/chat/completions'; // Endpoint de l'API ChatGPT
  const headers = {
    'Content-Type': 'application/json',
    'Authorization': `Bearer ${api_key}`
  };

  // Données de la requête
  const data = {
    'messages': [{'role': 'system', 'content': 'You are a helpful assistant.'}, {'role': 'user', 'content': prompt}],
    'max_tokens': 1024
  };

  // Effectuer la requête POST à l'API ChatGPT
  const response = await fetch(endpoint, {
    method: 'POST',
    headers: headers,
    body: JSON.stringify(data)
  });

  // Vérifier la réponse de l'API
  if (!response.ok) {
    throw new Error('Erreur lors de la requête à l\'API ChatGPT');
  }

  // Récupérer la réponse en tant qu'objet JSON
  const result = await response.json();
  return result.choices[0].message.content;
}

// Exemple d'utilisation
const prompt = "Bonjour, comment ça va ?";
generateChatGPTMessage(prompt)
  .then(message => {
    console.log('Réponse de ChatGPT :', message);
  })
  .catch(error => {
    console.error('Erreur :', error);
  });
Dans cet exemple, nous utilisons la fonction fetch pour effectuer une requête POST à l'API ChatGPT de OpenAI en passant la clé d'API dans les en-têtes de la requête. Les données de la requête sont fournies sous forme d'objet JSON, avec le prompt du message que vous souhaitez envoyer à ChatGPT. La réponse de l'API est ensuite traitée et le message généré par ChatGPT est extrait du résultat. Assurez-vous d'avoir configuré correctement votre clé d'API dans le fichier de configuration PHP comme mentionné précédemment.
